前言

To be honest，很多题解给人一种斜率优化就必须用单调队列的错觉。

Actually，单调队列只是一种维护凸包的方式。有一定局限性的，本文旨在提供一种通法的同时让读者认识到各种斜率优化的优点与局限性以及其适用范围。

需要掌握的知识：

数组，赋值，循环，递归，STL map （平衡树），线段树。

如有错误请直接喷作者[https://www.luogu.com.cn/user/367687]。

本质

斜率优化本质是将 dp 方程转化为最大/小化斜率的问题，然后通过维护凸包等方式求解最优值。

对于一个 dp 方程

dp[i] = min/max{dp[j]+w(i,j)}

将原方程去掉 min/max 函数后转换为形如 ... 的 dp 方程。

Y(j)=K(i)∗X(j)+f(i)+C(i) 

那么对于每一个点都有对应的横纵坐标，用斜率为 K(i) 的直线去经过点 j 时得到的截距就是一个与 i 相关的值，

由于原方程中要求最大/最小值，在平面内就需要最大/最小化截距。这时只需要维护一个凸包即可。

常用方法

单调队列

最基础的维护方式。

FOR(i,1,n)
{
    while(h<t&&slope(q[h],q[h+1])<=K(i))h++;
    int j=q[h];f[i]=f[j]+(s[i]-s[j])*T[i]+(s[n]-s[j])*S;
    while(h<t&&slope(q[t-1],q[t])>=slope(q[t],i))t--;
    q[++t]=i;
}

李超树

比较全能并且性价比很高的维护方式。（横坐标得是整数才可以）

xxxxxxxxxx
int n,h[maxn],w[maxn],c,f[maxn];
int s[maxn<<2],u,v;
double k[maxn],b[maxn];
inline int g(int x,int o){return b[o]+k[o]*x;}
void upd(int k,int l,int r,int t)
{
    if(l==r)return (g(l,t)<g(l,s[k]))&&(s[k]=t),void();
    int mid=l+r>>1;
    if(g(mid,t)<g(mid,s[k]))swap(t,s[k]);
    if(g(l,t)<g(l,s[k]))upd(k<<1,l,mid,t);
    else if(g(r,t)<g(r,s[k]))upd(k<<1|1,mid+1,r,t);
}
int qry(int k,int l,int r)
{
    if(l==r)return g(u,s[k]);
    int mid=l+r>>1;
    return gmin(g(u,s[k]),u<=mid?qry(k<<1,l,mid):qry(k<<1|1,mid+1,r));
}

参考 CF70D Professor′s task

用 map 维护凸包，每次加入新点的时候不断在左右删点维护凸性即可。

查询最值时在凸包上二分，注意不要越界。

由于每个点最多进一次出一次，加上 map 的复杂度，维护过程的复杂度是 O(nlogn) 。

二分的复杂度是 O(nlog2n)。

这样一来，只要找到了合适的 dp 方程我们都可以在 nlog2(n) 内通过题目。

然而实际上，我们并不一定要把这种暴力方法写出来，利用题目信息找到单调性能更方便快速地解题。

参考代码：（还是比较短的，注意特判较多）

xxxxxxxxxx
#define mii map<int,int>
#define miit map<int,int>::iterator
int n;
mii s;
inline int det(int a,int b,int c,int d){return a*d-b*c;}
inline bool Qry(int x,int y)
{
    miit it=s.lower_bound(x);
    if(it==s.end())return 0;
    if(it->first==x)return y>=it->second;
    if(it==s.begin())return 0;
    miit jt=it;jt--;
    return (det(jt->first-x,jt->second-y,it->first-x,it->second-y)>=0);
}
inline bool Rmv(miit it)
{
    miit jt=it,kt=it;
    if(jt==s.begin())return 0;
    jt--,kt++;
    if(kt==s.end())return 0;
    if(det(it->first-jt->first,it->second-jt->second,kt->first-it->first,kt->second-it->second)<0){s.erase(it);return 1;}
    return 0;
}
inline void Ins(int x,int y)
{
    if(Qry(x,y))return;s[x]=y;
    miit it=s.lower_bound(x);miit jt=it;
    if(jt!=s.begin()){jt--;while(Rmv(jt++))jt--;}
    if(++jt!=s.end()){while(Rmv(jt--))jt++;}
}
inline miit Find(double k)
{
    int l=0,r=(s.rbegin()->first);
    miit it,jt;
    while(l<r)
    {
        int mid=l+r>>1;
        it=s.lower_bound(mid);
        jt=it;jt++;if(jt==s.end())it--,jt--;
        if(1.0*(it->second-jt->second)/(it->first-jt->first)<=k)l=mid+1;
        else r=mid;
    }
    return s.lower_bound(l+r>>1);
}
